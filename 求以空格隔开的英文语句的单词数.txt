#include <stdio.h>
#include <string.h>

int main(void){
    int c = 0,n,i;
    char a[200];
   
    printf("请输入一段以空格隔开的英文语句：\n");
    gets(a);
    n = strlen(a);
    
    i = 0;
    while(i <= n-1){
        if(a[i] != ' ' && a[i+1] = ' ')
            c++;
        i++;
    }
    printf("该段英文语句所包含%d个单词",c);
  
    return 0;
}
