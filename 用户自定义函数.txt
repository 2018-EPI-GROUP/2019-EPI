5，编写一个求两个数最小公倍数的函数，并求出两个数的最小公倍数，示例代码如下：
#include <stdio.h>
int h(int x,int y){
    int r,p;
    p=x*y;
    while((r=y%x)!=0){
        y = x;
        x = r;
    }
    p=p/x;
    return (p);
}
int main(){
    int a,b,c;
    scanf("%d %d",&a,&b);
    c=h(a,b);
    printf("c = %d",c);
    return 0;
}
