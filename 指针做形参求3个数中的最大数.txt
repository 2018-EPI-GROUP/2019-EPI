#include <stdio.h>
void find(int a,int b,int c,int *p){
    int m;
    if(a > b)
    m = a;
    else m = b;
    if(m < c)
    m = c;
      
    *p = m;
    return;
}
int main(void){
    int x,y,z,max;
    printf("请输入3个整数：\n");
    scanf("%d%d%d",x,y,z);
    find(x,y,z,max);
    printf("最大数为%d\n",max);
    return 0;
}
