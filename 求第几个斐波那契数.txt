#include <stdio.h>
int main(void){
    int num1 = 1,num2 = 1;
	int i,n,ret;
	
	printf("请输入一个整数：\n");
	scanf("%d",&n);
	
	if(n == 1 || n == 2)
	    ret = 1;
	else if(n > 2){
		for(i = 0;i < n-2;i++){
			ret  = num1 + num2;
			num2 = num1;
			num1 = ret;
		}
	}	
	printf("第%d个斐波那契数为:%d",n,ret);
	
	return 0;
	
}
