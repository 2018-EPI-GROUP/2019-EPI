#include <stdio.h>
void fun(int *p){
    int i;
    for(i = 0;i <= 9;i++)
        *(p + i) = *(p + i)*10;
    return;
}
int main(void){
    int a[10],i;
    printf("请输入10个整数：\n");
    for(i = 0;i <= 9;i++)
        scanf("%d",&a[i]);
    fun(a);
    for(i = 0;i <= 9;i++)
        printf("%d\t",a[i]);
    return 0;
}
