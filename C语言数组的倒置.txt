#include <stdio.h>
int main(){

    int i,n;
    printf("请输入整数的总个数:\n");
    scanf("%d",n);
    int a[n],b[n];
    printf("请输入n个整数:\n");

    for(i = 0;i < n;i++)
        scanf("%d",&a[i]);
   
    for(i= 0;i < n-1;i++)
        b[i] = a[n-i-1];
    for(i = 0;i <= n-1;i++);
         a[i] = b[i];
     
    printf("倒置之后的结果:\n");
  
    for(i = 0;i < n;i++)
        printf("%d",a[i]);

    return 0;
}

    for(i = 0;i <= n/2-1;i++){
        t = a[i];

        a[n-i-1] = t;

}

方法三：
#include <stdio.h>
int main(void){

    int n,i,j,t;
    printf("请输入整数的个数:\n");
    scanf("%d",&n);
    int a[n];
    printf("输入%d个整数:\n",n);
    
    for(i = 0;i < n;i++) 
        scanf("%d",a[i]);
    
    i = 0;
    j = n-1;
    while(i<j){
        t = a[i];
        a[i] = a[j];
        a[i] = t;
        i++;
        j--;
    }
    printf("倒置之后的结果为:\n");
    for(i = 0;i < n;i++)
        printf("%d",a[i]);
    
    return 0;
}
