#include <stdio.h>
#define N 10
int main(void){
    int i,j,t;
    int a[N];
    printf("请输入%d个整数:\n",N);
    
    for(i = 0;i < N-1;i++){
        scanf("%d",&a[i]);
    }

    for(i = 0;i < N-2;i++){
        for(j = i + 1;j < N-1;j++){
            if(a[i] < a[j]){
                t = a[j];
                a[j] = a[i];
                a[i] = t;
            }
        }
    }
    printf("排序之后的结果是:\n");
    for(i = 0;i < N-1;i++){
        printf("%d",a[i]);
    }
    
    return 0;
}
