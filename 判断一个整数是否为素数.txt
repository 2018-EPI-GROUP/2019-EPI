#include <stdio.h>
int main(void){
	int a = 0,n = 0;
	printf("Please input an integer:\n");
	scanf("%d",&n);
	
	for(int i = 2;i < n;i++){
		if(n % i ==0){
		    a++;
		}
	}
	if(a == 0){
	printf("%d is a prime.\n",n);
	}
	else{
    printf("%d is not a prime.\n",n);
	}

    return 0;
}
